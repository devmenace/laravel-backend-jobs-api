Clone repository
``
git clone git@github.com:devmenace/laravel-backend-jobs-api.git
``

Install dependencies
``
composer install && npm install
``

Create config file and add DB configs
``
cp .env.example .env
``

``
DB_CONNECTION=mysql
DB_HOST=127.0.0.1
DB_PORT=3306
DB_DATABASE=laravel
DB_USERNAME=demo
DB_PASSWORD=demo
